CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(TestNSSteadyDD)

# Build the executable
ADD_EXECUTABLE(testNSSteadyDD ${CMAKE_PROJECT_DIR}main.cpp)

# Link the executable to the petsc anf femttu libs
TARGET_LINK_LIBRARIES(testNSSteadyDD femus)
TARGET_LINK_LIBRARIES(testNSSteadyDD ${PETSC_LIBRARIES})
TARGET_LINK_LIBRARIES(testNSSteadyDD b64)
TARGET_LINK_LIBRARIES(testNSSteadyDD jsoncpp)
TARGET_LINK_LIBRARIES(testNSSteadyDD ${ADEPT_LIBRARIES})

IF(MPI_FOUND)
  TARGET_LINK_LIBRARIES(testNSSteadyDD ${MPI_EXTRA_LIBRARY})
ENDIF(MPI_FOUND)

IF(FPARSER_FOUND)
  TARGET_LINK_LIBRARIES(testNSSteadyDD ${FPARSER_LIBRARY})
ENDIF(FPARSER_FOUND)

ADD_TEST(NAME testNSSteadyDD COMMAND valgrind --xml=yes --xml-file=${CMAKE_BINARY_DIR}/unittests/testNSSteadyDD.xml ./testNSSteadyDD)
