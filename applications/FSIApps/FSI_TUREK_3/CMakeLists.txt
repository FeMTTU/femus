CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(FSI_TUREK_3)

# Build the executable
ADD_EXECUTABLE(fsitrk ${CMAKE_PROJECT_DIR}main.cpp)

# Link the executable to the petsc anf femttu libs
TARGET_LINK_LIBRARIES(fsitrk femus)
TARGET_LINK_LIBRARIES(fsitrk ${PETSC_LIBRARIES})
TARGET_LINK_LIBRARIES(fsitrk b64)
TARGET_LINK_LIBRARIES(fsitrk jsoncpp)
TARGET_LINK_LIBRARIES(fsitrk ${ADEPT_LIBRARIES})

IF(FPARSER_FOUND)
  TARGET_LINK_LIBRARIES(fsitrk ${FPARSER_LIBRARY})
ENDIF(FPARSER_FOUND)

IF(MPI_FOUND)
  TARGET_LINK_LIBRARIES(fsitrk ${MPI_EXTRA_LIBRARY})
ENDIF(MPI_FOUND)
